const baseVertexShaderPatch = `
  uniform float uTime;
  varying vec2 vUvBase;
`;
const baseVertexShaderMainPatch = `
  vUvBase = uv;
  vec3 transformed = vec3(position);
  if (uTime > 0.0) {
    transformed.y += sin(transformed.x * 8.0 + uTime) * 0.008 * sin(transformed.z * 8.0);
  }
  gl_Position = projectionMatrix * modelViewMatrix * vec4(transformed, 1.0);
`;
const sauceFragmentShader = `
  uniform float uTime;
  varying vec2 vUv;
  float hash(vec2 p) { return fract(sin(dot(p, vec2(12.9898, 78.233))) * 43758.5453); }
  float noise(vec2 p) {
    vec2 i = floor(p); vec2 f = fract(p);
    vec2 u = f * f * (3.0 - 2.0 * f);
    return mix(mix(hash(i + vec2(0.0,0.0)), hash(i + vec2(1.0,0.0)), u.x),
               mix(hash(i + vec2(0.0,1.0)), hash(i + vec2(1.0,1.0)), u.x), u.y);
  }
  float fbm(vec2 p) {
    float f = 0.0; float w = 0.5;
    for(int i = 0; i < 4; i++) { f += w * noise(p); p *= 2.0; w *= 0.5; }
    return f;
  }
  void main() {
    float n = fbm(vUv * 5.0 + uTime * 0.2);
    vec3 c1 = vec3(0.54, 0.10, 0.10);
    vec3 c2 = vec3(0.75, 0.15, 0.12);
    gl_FragColor = vec4(mix(c1, c2, n), 1.0);
  }
`;
const sauceVertexShader = `
  varying vec2 vUv;
  void main() { vUv = uv; gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0); }
`;

function DynamicCheeseLayer() {
  const geomRef = useRef<THREE.PlaneGeometry>(null);
  useEffect(() => {
    if (geomRef.current) {
      const pos = geomRef.current.attributes.position;
      let seed = 123;
      const rand = () => { seed = (seed * 16807) % 2147483647; return (seed - 1) / 2147483646; };
      for (let i = 0; i < pos.count; i++) {
        const x = pos.getX(i), y = pos.getY(i);
        if (Math.sqrt(x * x + y * y) < 1.6) pos.setZ(i, pos.getZ(i) + (rand() - 0.5) * 0.04);
      }
      pos.needsUpdate = true;
      geomRef.current.computeVertexNormals();
    }
  }, []);
 
  const cheeseMaterial = useMemo(() => {
    const mat = new THREE.MeshStandardMaterial({ color: '#F5E6A3', roughness: 0.2, alphaTest: 0.5 });
    mat.onBeforeCompile = (shader) => {
      shader.fragmentShader = shader.fragmentShader.replace(
        '#include <clipping_planes_fragment>',
        `#include <clipping_planes_fragment>\nif (length(vUv - 0.5) > 0.49) discard;`
      );
    };
    return mat;
  }, []);
 
  return (
    <mesh position={[0, 0.14, 0]} rotation={[-Math.PI / 2, 0, 0]}>
      <planeGeometry ref={geomRef} args={[3.2, 3.2, 32, 32]} />
      <primitive object={cheeseMaterial} attach="material" />
    </mesh>
  );
}
 
// Static toppings layer for menu cards (non-interactive)
function StaticToppingsLayer({ toppingIds }: { toppingIds: ToppingID[] }) {
  const instances = useMemo(() => {
    const nodes: React.ReactNode[] = [];
    let globalSeed = 1000;
    toppingIds.forEach((id) => {
      const tDef = TOPPINGS.find(t => t.id === id);
      if (!tDef) return;
      let count = 8;
      if (tDef.mesh === 'sphere') count = 20;
      else if (tDef.mesh === 'sliver') count = 12;
      const positions = distributeToppings(count, 1.4, globalSeed++);
      positions.forEach(([x, z], i) => {
        const y = Math.abs(x * z) % 0.04 + 0.15;
        const rotY = (x + z) * Math.PI;
        let geom: React.ReactNode;
        if (tDef.mesh === 'disc') geom = <cylinderGeometry args={[0.12, 0.12, 0.02, 16]} />;
        else if (tDef.mesh === 'sliver') geom = <boxGeometry args={[0.1, 0.04, 0.3]} />;
        else if (tDef.mesh === 'chunk') geom = <boxGeometry args={[0.15, 0.12, 0.15]} />;
        else geom = <sphereGeometry args={[0.07, 12, 12]} />;
        const rot: [number, number, number] = tDef.mesh === 'disc' ? [Math.PI / 2, 0, rotY] : [Math.abs(x) % 0.4, rotY, Math.abs(z) % 0.4];
        nodes.push(<mesh key={`${id}-${i}`} position={[x, y, z]} rotation={rot}>{geom}<meshStandardMaterial color={tDef.color} roughness={0.6} /></mesh>);
      });
    });
    return nodes;
  }, [toppingIds]);
  return <group>{instances}</group>;
}
 
// Interactive animated toppings layer for the builder
function AnimatedToppingsLayer({ toppingIds }: { toppingIds: ToppingID[] }) {
  const [activeToppings, setActiveToppings] = useState<Map<ToppingID, { removing: boolean }>>(new Map());
  const prevIds = useRef<ToppingID[]>([]);
 
  useEffect(() => {
    const currentSet = new Set(toppingIds);
    const prevSet = new Set(prevIds.current);
 
    setActiveToppings(prev => {
      const next = new Map(prev);
      // Mark removed toppings
      for (const id of Array.from(prevSet)) {
        if (!currentSet.has(id) && next.has(id)) {
          next.set(id, { removing: true });
        }
      }
      // Add new toppings
      for (const id of Array.from(currentSet)) {
        if (!prev.has(id)) {
          next.set(id, { removing: false });
        }
      }
      return next;
    });
    prevIds.current = [...toppingIds];
  }, [toppingIds]);
 
  const handleRemoved = useCallback((id: ToppingID) => {
    setActiveToppings(prev => {
      const next = new Map(prev);
      next.delete(id);
      return next;
    });
  }, []);
 
  const entries = Array.from(activeToppings.entries());
  let staggerBase = 0;
 
  return (
    <group>
      {entries.map(([id, state]) => {
        const delay = staggerBase;
        staggerBase += 0.08;
        return (
          <ToppingInstance
            key={id}
            toppingId={id}
            removing={state.removing}
            onRemoved={() => handleRemoved(id)}
            staggerBase={delay}
          />
        );
      })}
    </group>
  );
}
 
function BaseAndSauce({ isDynamic }: { isDynamic: boolean }) {
  const baseMaterialRef = useRef<THREE.MeshStandardMaterial>(null);
  const sauceUniforms = useMemo(() => ({ uTime: { value: 0 } }), []);
 
  useFrame(({ clock }) => {
    if (!isDynamic) return;
    if (baseMaterialRef.current && baseMaterialRef.current.userData.shader) {
      baseMaterialRef.current.userData.shader.uniforms.uTime.value = clock.getElapsedTime();
    }
    sauceUniforms.uTime.value = clock.getElapsedTime();
  });
 
  const customBaseMaterial = (shader: { uniforms: Record<string, { value: number }>; vertexShader: string }) => {
    shader.uniforms.uTime = { value: isDynamic ? 0 : -1.0 };
    shader.vertexShader = baseVertexShaderPatch + shader.vertexShader;
    shader.vertexShader = shader.vertexShader.replace('#include <begin_vertex>', baseVertexShaderMainPatch);
    if (baseMaterialRef.current) baseMaterialRef.current.userData.shader = shader;
  };
 
  return (
    <group>
      <mesh>
        <cylinderGeometry args={[1.8, 1.9, 0.18, 64]} />
        <meshStandardMaterial ref={baseMaterialRef} color="#C4955A" roughness={0.85} onBeforeCompile={customBaseMaterial} />
      </mesh>
      <mesh>
        <cylinderGeometry args={[1.82, 1.95, 0.22, 64]} />
        <meshStandardMaterial color="#8B5E2A" roughness={0.95} />
      </mesh>
      <mesh position={[0, 0.11, 0]} rotation={[-Math.PI / 2, 0, 0]}>
        <circleGeometry args={[1.62, 64]} />
        <shaderMaterial uniforms={sauceUniforms} vertexShader={sauceVertexShader} fragmentShader={sauceFragmentShader} />
      </mesh>
    </group>
  );
}
 
function SteamParticles() {
  const count = 80;
  const meshRef = useRef<THREE.Points>(null);
  const [positions, speeds] = useMemo(() => {
    const pos = new Float32Array(count * 3);
    const spd = new Float32Array(count);
    for (let i = 0; i < count; i++) {
      const r = Math.random() * 1.5;
      const theta = Math.random() * Math.PI * 2;
      pos[i * 3] = r * Math.cos(theta);
      pos[i * 3 + 1] = Math.random() * 2.0;
      pos[i * 3 + 2] = r * Math.sin(theta);
      spd[i] = 0.2 + Math.random() * 0.3;
    }
    return [pos, spd];
  }, []);
 
  useFrame((_, delta) => {
    if (meshRef.current) {
      const posAttr = meshRef.current.geometry.attributes.position;
      for (let i = 0; i < count; i++) {
        let y = posAttr.getY(i);
        y += speeds[i] * delta;
        if (y > 2.0) y = 0.0;
        posAttr.setY(i, y);
      }
      posAttr.needsUpdate = true;
    }
  });
 
  return (
    <points ref={meshRef}>
      <bufferGeometry>
        <bufferAttribute attach="attributes-position" count={positions.length / 3} array={positions} itemSize={3} />
      </bufferGeometry>
      <pointsMaterial size={0.04} color="#FFFFFF" opacity={0.15} transparent depthWrite={false} blending={THREE.AdditiveBlending} />
    </points>
  );
}
 
function StaticCamera({ thumbnailMode }: { thumbnailMode?: boolean }) {
  const { camera } = useThree();
  useEffect(() => {
    if (thumbnailMode) {
      camera.position.set(0, 4, 0);
    } else {
      camera.position.set(2.5, 2.5, 2.5);
    }
    camera.lookAt(0, 0, 0);
  }, [camera, thumbnailMode]);
  return null;
}
 
// FOV pulse on topping changes
function FovPulse({ toppingIds }: { toppingIds: ToppingID[] }) {
  const { camera } = useThree();
  const prevCount = useRef(toppingIds.length);
  const pulseTarget = useRef<number | null>(null);
  const pulsePhase = useRef<'down' | 'up'>('down');
 
  useEffect(() => {
    if (toppingIds.length !== prevCount.current) {
      pulseTarget.current = 42;
      pulsePhase.current = 'down';
    }
    prevCount.current = toppingIds.length;
  }, [toppingIds]);
 
  useFrame((_, delta) => {
    if (pulseTarget.current === null) return;
    const cam = camera as THREE.PerspectiveCamera;
    if (pulsePhase.current === 'down') {
      cam.fov = THREE.MathUtils.lerp(cam.fov, 42, 10 * delta);
      if (Math.abs(cam.fov - 42) < 0.3) pulsePhase.current = 'up';
    } else {
      cam.fov = THREE.MathUtils.lerp(cam.fov, 45, 8 * delta);
      if (Math.abs(cam.fov - 45) < 0.2) {
        cam.fov = 45;
        pulseTarget.current = null;
      }
    }
    cam.updateProjectionMatrix();
  });
 
  return null;
}
 
// Animated pizza scale based on size
const SIZE_SCALE: Record<Size, number> = { small: 1.0, medium: 1.1, large: 1.22 };
 
function AnimatedPizzaGroup({ size, children }: { size: Size; children: React.ReactNode }) {
  const groupRef = useRef<THREE.Group>(null);
  const targetScale = SIZE_SCALE[size];
 
  useFrame((_, delta) => {
    if (groupRef.current) {
      const current = groupRef.current.scale.x;
      const next = THREE.MathUtils.lerp(current, targetScale, 6 * delta);
      groupRef.current.scale.setScalar(next);
    }
  });
 
  return <group ref={groupRef}>{children}</group>;
}

export interface PizzaCanvasProps {
  toppingIds: ToppingID[];
  size: Size;
  interactive?: boolean;
  frameloop?: 'always' | 'demand';
  thumbnailMode?: boolean;
}
 
export default function PizzaCanvas({ toppingIds, size, interactive = false, frameloop = 'demand', thumbnailMode = false }: PizzaCanvasProps) {
  return (
    <Canvas
      frameloop={frameloop}
      camera={{ fov: 45, position: thumbnailMode ? [0, 4, 0] : [2.5, 2.5, 2.5] }}
      dpr={[1, interactive ? 2 : (thumbnailMode ? 1 : 1.5)]}
      gl={{ antialias: true, powerPreference: 'high-performance' }}
    >
      {!interactive && <StaticCamera thumbnailMode={thumbnailMode} />}
 
      {interactive && (
        <>
          <OrbitControls
            autoRotate
            autoRotateSpeed={0.4}
            enableZoom={false}
            enablePan={false}
            minPolarAngle={Math.PI / 6}
            maxPolarAngle={Math.PI / 3}
          />
          <FovPulse toppingIds={toppingIds} />
        </>
      )}
 
      <ambientLight intensity={0.4} />
      <directionalLight position={[5, 8, 3]} intensity={1.5} color="#FFF5E0" castShadow />
      <pointLight position={[-3, 2, -2]} intensity={0.3} color="#E8540A" />
 
      <AnimatedPizzaGroup size={size}>
        <BaseAndSauce isDynamic={frameloop === 'always'} />
        <DynamicCheeseLayer />
        {interactive ? (
          <AnimatedToppingsLayer toppingIds={toppingIds} />
        ) : (
          <StaticToppingsLayer toppingIds={toppingIds} />
        )}
      </AnimatedPizzaGroup>
 
      {interactive && <SteamParticles />}
 
      <EffectComposer multisampling={0}>
        {frameloop === 'demand' ? (
          <Vignette darkness={0.3} offset={0.3} />
        ) : (
          <>
            <Bloom luminanceThreshold={0.8} intensity={0.4} />
            <DepthOfField focusDistance={0} focalLength={0.08} bokehScale={2} />
            <Vignette darkness={0.3} offset={0.3} />
            <Noise opacity={0.03} />
          </>
        )}
      </EffectComposer>
    </Canvas>
  );
}
